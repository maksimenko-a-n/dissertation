\begin{frame}{Пример: задача о рюкзаке}
\begin{block}{Задача о рюкзаке}
	\begin{description}
	\item[\textbf{Дано:}] множество предметов $E = \{e_1, \dots, e_n\}$,\\
	стоимости предметов $\vec{c} = (c_1, \dots, c_n)$,\\
	размеры предметов $\vec{a} = (a_1, \dots, a_n)$,\\
	размер рюкзака $A$.
	\item[\textbf{Найти:}] подмножество $S \subseteq E$ такое, что\\
		\(\displaystyle\sum_{e_i \in S} c_i \to \max\), \quad
		при условии \(\displaystyle\sum_{e_i \in S} a_i \le A\).
	\end{description}
\end{block}
\begin{block}{Геометрическая интерпретация}
	\begin{description}
	\item[\textbf{Дано:}] $n \in \N$, векторы $\vec{c}, \vec{a} \in \R^n$ и число $A \in \R$.
	\item[\textbf{Найти:}] 0/1-вектор $\vec{x} \in \{0,1\}^n$ такой, что\\
	\(\langle\vec{x}, \vec{c}\rangle \to \max\), \quad
	при условии \(\langle\vec{x}, \vec{a}\rangle \le A\).
	\end{description}
	\(X(\vec{a}, A) = \{\vec{x} \in \{0,1\}^n \mid  \langle \vec{a}, \vec{x}\rangle \le A\}\) --- допустимые решения,\\
	\(\alert{\KNAP (\vec{a}, A)} \coloneqq \conv (X(\vec{a}, A))\) --- \alert{многогранник задачи о рюкзаке}
\end{block}
\end{frame}


\begin{frame}{Пример: задача о кратчайшем пути}
\begin{block}{Задача о кратчайшем пути в орграфе}
\begin{description}
	\item[Дано:] полный орграф $D_n = ([n],A)$, $[n] \coloneqq \{1,\dots,n\}$,\\
	вектор длин дуг $\vec{c} \in \R^A$.
	\item[Найти:] кратчайший путь в $D_n$ из вершины $1$ в $n$.
\end{description}
\end{block}

\begin{block}{Геометрическая интерпретация}
	\begin{description}
		\item[\textbf{Дано:}] $n \in \N$ и вектор $\vec{c} \in \R^A$, $|A| = n(n-1)$.
		\item[\textbf{Найти:}] $\vec{x} \in \{0,1\}^A$ такой, что\\
		\(\langle\vec{x}, \vec{c}\rangle \to \max\),\\
		\(\vec{x}\) --- характеристический вектор пути в $D_n$ из $1$ в $n$.
	\end{description}
\end{block}
\begin{block}{Многогранник орпутей}
	\(\alert{\Dipath (n)} \coloneqq \conv \Bigl\{\vec{x} \in \{0,1\}^A \Bigm| \vec{x} \text{ --- путь в $D_n = ([n],A)$ из 1 в $n$}\Bigr\}\)
\end{block}
\end{frame}

\begin{frame}{Пример: булево квадратичное программирование}
\begin{block}{Булево квадратичное программирование}
\begin{description}
	\item[Дано:] число $n$ булевых переменных $x_i \in \{0,1\}$, $i \in [n]$;\\ матрица коэффициентов $\vec{c} = (c_{ij}) \in \R^{n \times n}$.
	\item[Найти:] $\max \displaystyle\sum_{i,j \in [n]} c_{ij} x_i x_j$.
\end{description}
\end{block}

\begin{block}{Геометрическая интерпретация}
	\begin{description}
		\item[\textbf{Дано:}] $n \in \N$ и матрица $\vec{c} \in \R^{n \times n}$.
		\item[\textbf{Найти:}] $\vec{x} = (x_{ij}) \in \{0,1\}^{n \times n}$ такой, что\\
		\(\langle\vec{x}, \vec{c}\rangle \to \max\),
		при условии \alert{$x_{ij} = x_{ii} x_{jj}$}.
	\end{description}
\end{block}
	
\begin{block}{Булев квадратичный многогранник}
	\[\alert{\BQP(n)} \coloneqq \conv \Bigl\{ \vec{x} = (x_{ij}) \in \{0,1\}^{n \times n} \Bigm| x_{ij} = x_{ii} x_{jj}, \ i,j\in[n] \Bigr\}\]
\end{block}
\end{frame}

\begin{frame}{Терминология}
\begin{block}{Общие черты таких задач}
	\begin{itemize}
		\item Линейная целевая функция \(\langle\vec{x}, \vec{c}\rangle \to \max\).
		\item Оптимизация на множестве $X \subseteq \{0,1\}^d$,\\ удовлетворяющем некоторым ограничениям\\
		(\(\langle\vec{x}, \vec{a}\rangle \le A\) для задачи о рюкзаке,
		$x_{ij} = x_{ii} x_{jj}$ для задачи BQP).
	\end{itemize}
\end{block}
\begin{block}{Понятия и обозначения}
	\begin{itemize}
		\item $\vec{c} \in \R^d$ "--- \alert{целевой вектор},
		\item $d \in \N$ "--- \alert{размерность задачи},
		\item $X \subseteq \{0,1\}^d$ "--- \alert{множество допустимых решений},
		\item входные данные, от которых зависит $X$, "--- \alert{код задачи}\\
		($n$, $\vec{a}$ и $A$ для задачи о рюкзаке, $n$ "--- для других двух примеров)
	\end{itemize}
\end{block}
\end{frame}


\begin{frame}{Линейная задача комбинаторной оптимизации}
	\vspace*{-\fill}
	\begin{block}{Определение}
		\textbf{Линейная задача комбинаторной оптимизации (ЛЗКО)} "---\\
		это~пара функций, аргументом которых является \alert{код задачи $I$}:
		\begin{enumerate}
			\item \alert{Размерность} задачи $\alert{d} = d(I) \in \N$.
			\item \alert{Предикат допустимости} $\alert{g(\vec{x}, I)} \in \{\text{ложь}, \text{истина}\}$, где $\vec{x} \in \{0,1\}^d$.
		\end{enumerate}
		\medskip
		\underline{Входные данные:} код задачи $I$ и \alert{целевой вектор} $\alert{\vec{c}} \in \Z^d$.
		
		\medskip
		\underline{Цель:} \alert{$\langle\vec{c},\vec{x}\rangle \to \max$} при условии $\vec{x} \in \{0,1\}^d$ и $g(\vec{x}, I) = \text{истина}$.
		%найти оптимальное решение
		%\[\vec{x^*} = \arg\max_{\vec{x} \in X} \langle\vec{c},\vec{x}\rangle,\]
	\end{block}
\begin{block}{Многогранники задачи}
	$X(I) \coloneqq \{\vec{x} \in \{0,1\}^d \mid g(\vec{x}, I)\}$ "--- множество допустимых решений.\\
	$\alert{P(I)} \coloneqq \conv(X(I))$ --- \alert{многогранник задачи}.\\
	При произвольном выборе кода $I$ получаем \alert{семейство многогранников}.
\end{block}
\end{frame}

\begin{frame}{ЛЗКО и задача линейного программирования}
	\begin{block}{ЛЗКО}
		$\langle\vec{c},\vec{x}\rangle \to \max$ при условии \alert{$\vec{x} \in \conv(X(I))$},\\
		где %$X(I) \subseteq \{0,\dots,k\}^d$ "--- множество допустимых решений,\\
		$\conv(X(I))$ "--- многогранник задачи.
	\end{block}

	\begin{block}{Задача линейного программирования}
		$\langle\vec{c},\vec{x}\rangle \to \max$ при условии \alert{$A \vec{x} \le \vec{b}$},
		где $\vec{x} \in \R^d$, $A \in \Z^{n \times d}$, $\vec{b} \in \Z^n$.
	\end{block}
	\begin{block}{Способы описания выпуклого многогранника}
		\begin{description}
		\item[V-описание:] выпуклая оболочка конечного множества точек.
		\item[H-описание:] пересечение полупространств.
		\end{description}
		\textbf{Пример} <<0/1-куб>>: $\conv\left(\{0,1\}^d\right) = \{\vec{x} \in \R^d \mid \vec{0} \le \vec{x} \le \vec{1}\}$.
	\end{block}
\end{frame}

\begin{frame}{Смежные решения}
	\begin{block}{Определение}
		Для задачи линейной оптимизации на конечном множестве $X \subset \Z^d$\\
		пару точек $\vec{x}, \vec{y} \in X$ назовем \alert{смежными},
		если $\exists$ $\vec{c_{xy}} \in \Z^d$ такой, что 
		\[\langle\vec{c_{xy}}, \vec{x}\rangle = \langle\vec{c_{xy}}, \vec{y}\rangle > \langle\vec{c_{xy}}, \vec{z}\rangle \qquad 
		\forall \vec{z} \in X \setminus \{\vec{x}, \vec{y}\}.\]
		Смежность $\vec{x}$ и $\vec{y}$ означает, что $\vec{x}$ и $\vec{y}$ являются вершинами многогранника $\conv(X)$ и образуют \alert{ребро} этого многогранника.
	\end{block}
	\begin{block}{Наблюдение}
		Алгоритм для линейной оптимизации на $X$ должен быть чувствителен к небольшим изменениям вектора $\vec{c_{xy}}$, так как оптимальное решение может меняться с $\vec{x}$ на $\vec{y}$ и обратно.
		
		Таким образом, наличие ребер (1-граней) между вершинами многогранника влияет на структуру алгоритма.
	\end{block}
\end{frame}

\begin{frame}{Диаметр графа многогранника}
	\begin{block}{Классическая мотивация}

\begin{frame}{Матрица инциденций вершин-гиперграней}
	\begin{block}{Определение}

\begin{frame}{Число прямоугольного покрытия}
	\begin{block}{Число прямоугольного покрытия матрицы инциденций $M$}
		\centering
		\(\alert{\rc(M)} \coloneqq \min\{\text{\# 0-прямоугольников для покрытия всех нулей в $M$}\}\)
	\end{block}
	
	\begin{block}{Теорема (Yannakakis '91)}
		Сложность расширения многогранника $P$ оценивается снизу\\
		числом прямоугольного покрытия матрицы инциденций $M$:\\
		\centerline{\(\rc(M) \le \xc(P)\).}
	\end{block}
	\begin{block}{Основные соотношения}
		Пусть $M$~--- матрица инциденций вершин"=гиперграней многогр-ка $P$,\\
		а $f(P)$~--- число его граней всех размерностей. Тогда
		\centerline{\(\dim(P) < \log_2 f(P) \le \alert{\rc(M) \le \xc(P)} \le \min\{\vertices(P), \facet(P)\}\)}
	\end{block}
% При этом есть примеры многогранников, для которых кликовое число меньше, чем $\dim(P)$, и есть примеры, для которых оно больше, чем правая часть этого неравенства
% Для диаметра графа правое неравенство, скорее всего, выполнено всегда (гипотеза Хирша), а левое --- нет (для многогранников с полным графом).
\end{frame}

\begin{frame}{Идея: сравнение многогранников}
\begin{block}{Сравнение многогранников: $p \le q$}
	Если многогранник $p$ аффинно эквивалентен многограннику $q$ или некоторой его грани, то пишем $p \le q$ ($p$ не сложнее $q$).
\end{block}
\begin{block}{Сравнение семейств многогранников: $P \propto_A Q$}

	Для каждого $p \in P$ найдется $q \in Q$ такой, что $p \le q$.
%	$p$ аффинно эквивалентен некоторой грани многогранника $q$.
\end{block}

\begin{frame}{Аффинная сводимость многогранников задач}
	\begin{block}{Определение}
		Семейство $P$ \alert{аффинно сводится} к $Q$ (обозначение: \alert{$P \propto_A Q$}),\\ если найдутся полиномиально вычислимые: %(относительно размера многогранника $p\in P$):
		\begin{enumerate}
			\item 
			\alert{Преобразование $\tau$ кода $I$} каждого многогранника $p = p(I)\in P$ \alert{в~код $I'$} многогранника $q = q(I') \in Q$.
			\item 
			Алгоритм построения для каждого кода $I$ \alert{аффинного отображения} 
			\[
			\alert{\alpha}\from \R^d \to \R^{d'}, \qquad d = d(I), \quad d' = d'(\tau(I)),
			\]
			такого, что образ \alert{$\alpha(p(I))$ аффинно эквивалентен $p(I)$\\ 
				и является гранью многогранника $q(\tau(I))$}.
		\end{enumerate}
	\end{block}
\end{frame}

  \begin{block}{Свойства}
	Пусть $P \propto_A Q$. 
	Тогда многогранники семейства $Q$ наследуют от $P$: % следующие свойства:
	
	\begin{enumerate}
		\item 
		сверхполиномиальное число вершин, граней, гиперграней 
		\item 
		сверхполиномиальное кликовое число графа многогранника
		\item 
		NP-полноту проверки несмежности вершин
		\item 
		сверхполиномиальную сложность расширения
		\item 
		сверхполиномиальность числа прямоугольного покрытия
		\item 
		\dots
	\end{enumerate}
\end{block}
\end{frame}

	%			Т.\,е. не~следует надеяться на~существование алгоритма такого типа для~задачи коммивояжера.
	%\end{quote}

\begin{frame}{Грань с NP-полным критерием несмежности}
	\begin{alertblock}{Теорема (\Maks '12)}%<2->
		Все вышеперечисленные многогранники содержат в~качестве граней\\ многогранники двойных покрытий $\Matsui$,\\
		для которых задача проверки на~несмежность NP-полна.
	\end{alertblock}
	\begin{alertblock}{}
		Не для всех многогранников NP-трудных задач это верно!
	\end{alertblock}
	\begin{block}{Теорема (Грешнев '84, Белошевский '86, Barahona\&Mahjoub '86)}<2->
		Все вершины булева квадратичного многогранника попарно смежны.
	\end{block}
	\begin{block}{Следствие}<2->
		Аффинная сводимость $\Matsui \propto_A \BQP$ невозможна,
		так как среди многогранников $\Matsui$
		есть многогранники с~неполным графом.
	\end{block}
\end{frame}


\begin{frame}{Многогранники независимых множеств}
	\begin{block}{Определение}
	Пусть $G=(V,E)$ --- произвольный граф на $n = |V|$ вершинах.
	\alert{Многогранником независимых множеств} называется
	\centerline{\(
		\alert{\Stable(G)} \coloneqq \conv\bigl\{\vec{x}\in\{0,1\}^V \bigm| x_v + x_u \le 1 \quad \forall \{v,u\} \in E\bigr\}.
		\)}
	\end{block}
	\begin{block}{Теорема (Chv\'atal '75)}<2->
	%У этого многогранника имеется простой критерий смежности вершин.
	Пара вершин многогранника $\Stable(G)$ смежна тогда и~только тогда, когда симметрическая разность соответствующих независимых множеств индуцирует связный подграф в~$G$.
	\end{block}
	\begin{block}{Следствие}<2->
		Если $\text{NP} \ne \text{P}$, то сводимость $\Matsui \propto_A \Stable$ невозможна.
	\end{block}
	\begin{block}{Вопрос}<3->
		А если убрать условие $\text{NP} \ne \text{P}$?
	\end{block}
\end{frame}

\begin{frame}{Многогранники с простым критерием смежности}
	\begin{alertblock}{Теорема (\Maks '17)}
		\begin{enumerate}
			\item $\Stable \propto_A \Matsui$.
			\item Многогранники $\Matsui$ (за исключением отрезков)\\
			не могут быть гранями многогранника $\Stable(G)$ ни при каком $G$.\\
			Следовательно, $\Matsui \npropto_A \Stable$.
			%сводимость $\Matsui \propto_A \Stable$ невозможна.
		\end{enumerate}
	\end{alertblock}
	\begin{alertblock}{Теорема (\Maks '15)}<2->
		Многогранники независимых множеств эквивалентны относительно аффинной сводимости многогранникам следующих задач:\\
		упаковки и~разбиения множества, 3-сочетания, раскраска графа.
	\end{alertblock}
\end{frame}

\begin{frame}{Сверхполиномиальность характеристик}
	\begin{alertblock}{Теорема (\Maks '11-17)}
	Семейство булевых квадратичных многогранников $\{\BQP_n\}$\\
	аффинно сводится к каждому из перечисленных выше\\
	семейств многогранников труднорешаемых задач,
	а также к~многогранникам следующих задач: линейное упорядочивание,\\
	задача Штейнера в~графе и~некоторых других.\\
	\end{alertblock}
	\vspace{-2mm}
	\begin{alertblock}{}
	\textbf{Замечание:} аффинное свед\'ение в обратную сторону невозможно.
	\end{alertblock}
	\begin{block}{Теорема (Fiorini, Massar, Pokutta, Tiwary, de Wolf '11)}<2->
	Число прямоугольного покрытия и~сложность расширения
	
	для~булева квадратичного многогранника экспоненциальны.
	\end{block}
	\begin{block}{Теорема (Deza, Laurent, Poljak '92)}<3->
	Булев квадратичный многогранник $\BQP_n$ 3-смежностен.\\
	Следовательно, кликовое число графа совпадает с числом вершин $2^n$.
	\end{block}
\end{frame}


\begin{frame}{Универсальные $k$-смежностные грани}
	\begin{block}{Вопрос}
		Для каких $k\in \N$ булевы квадратичные многогранники содержат\\ $k$-смежностные грани со сверхполиномиальным числом вершин?
	\end{block}
	\begin{alertblock}{Теорема (\Maks '14)}<2->
		Для каждого $k\ge 4$ булевы квадратичные многогранники\\ 
		(а вместе с ними и все остальные, перечисленные выше семейства)\\
		содержат $k$-смежностные грани со~сверхполиномиальным \\
		(относительно размерности многогранника) числом вершин.
	\end{alertblock}
\end{frame}

\begin{frame}{Аналог теоремы Кука для многогранников}
	\begin{block}{Аффинная сводимость}
		Для каждого многогранника $p(I) \in P$ определен многогранник $q(I') \in Q$ такой, что $p(I)$ \alert{аффинно эквивалентен} грани $q(I')$.
	\end{block}
	\begin{block}{Расширенная аффинная сводимость}
		Для каждого многогранника $p(I) \in P$ определен многогранник $q(I') \in Q$ такой, что $p(I)$ \alert{является аффинным образом} грани $q(I')$.
	\end{block}
	\begin{alertblock}{Теорема (\Maks '12)}
		Пусть для семейства многогранников $\bigl\{\conv(X(I))\bigr\}$ задача проверки условия $\vec{x} \in X(I)$ (предикат допустимости) принадлежит классу NP.
		Тогда это семейство расширенно аффинно сводится к булевым квадратичным многогранникам.
	\end{alertblock}
\end{frame}

\begin{frame}{Сложность расширения $k$-смежностного многогранника}
\begin{block}{Наблюдение}
	Выше было показано, что многогранники NP-трудных задач\\
	$\forall k \in \N$	содержат $k$-смежностные грани\\
	со сверхполиномиальным числом вершин.\\
	Кроме того, сложность их расширения сверхполиномиальна.
\end{block}
\begin{block}{Факт}
	$k$-смежностные многогранники обладают\\ максимальным числом $i$-граней для $i \in [k]$\\ (при фиксированных размерности и числе вершин).
\end{block}
\begin{block}{Гипотеза}
	Наличие $k$-смежностных граней ($\forall k \in \N$) со сверхполиномиальным числом вершин является причиной высокой сложности расширения.
\end{block}
\end{frame}


\begin{frame}{Циклический многогранник}
	\begin{block}{Определение}
		Пусть $T = \{t_1, \dots, t_n\} \subseteq \R$, $t_1 < t_2 < \dots < t_n$.\\
		\alert{Циклическим многогранником} называется выпуклая оболочка
		\[\alert{\CP(d,T)} \coloneqq \conv\bigl\{(t, t^2, \dots, t^d) \bigm| t \in T\bigr\}, \qquad 2 \le d < n,\]
		$d$ --- размерность многогранника,
		$n$ --- число его вершин.
	\end{block}
	
	\begin{block}{Теорема, McMullen '70 (гипотеза, Motzkin '57)}
		Циклический многогранник~\alert{$\CP(d,T)$} для любого $k \in [d]$ \alert{обладает\\ максимальным числом $k$-граней} среди всех выпуклых многогранников той же размерности и с~тем~же числом вершин.
		
		В частности, он $\lfloor d/2\rfloor$-смежностен.
%		\\(т. е. любые $\lfloor d/2\rfloor$ вершин образуют грань).
	\end{block}
\end{frame}

\begin{frame}{Оптимизация на циклическом многограннике}
	\vspace{-1mm}
	\begin{block}{Максимум многочлена}
		Пусть $\vec{x} = (t, t^2, \dots, t^d)$, $t \in T \subseteq \R$, $\vec{c} \in \Z^d$.
		Тогда\\
		\centerline{$\langle \vec{c}, \vec{x}\rangle = c_1 t + c_2 t^2 + \dots + c_d t^d$.}
		Следовательно, задача $\langle \vec{c}, \vec{x}\rangle \to \max$ при $\vec{x} \in \CP(d,T)$
		%линейной оптимизации на многограннике $\CP(d,T)$
		эквивалентна нахождению максимума многочлена 
		$c_1 t + c_2 t^2 + \dots + c_d t^d$ при $t \in T$.
		%степени $d$, определенного на конечном множестве $T$.
	\end{block}
	\vspace{-1mm}
	\begin{block}{Теорема, Pan '96}
		Для нахождения корней многочлена степени $d$ с точностью $\varepsilon \in (0,1)$ (относительно радиуса области поиска) требуется порядка $\Theta(d^2 (d + \lvert\log \varepsilon\rvert))$ битовых операций.
	\end{block}
	\vspace{-1mm}
	\begin{block}{Следствие}
		Если $T = \Z \cap [a,b]$, то \alert{задача линейной оптимизации на $\CP(d,T)$ полиномиально разрешима} относительно $d$, $\log_2 (b+1-a)$\\
		и длины двоичной записи целевого вектора $\vec{c}$.
	\end{block}
\end{frame}

\begin{frame}{Циклический многоугольник: сложность расширения}
	\vspace{-15pt}
	\begin{minipage}[t]{0.58\textwidth}
		\begin{block}{}
			Пусть $T_n = \{t_1, \dots, t_n\} \subset \R$.
		\end{block}
		\begin{alertblock}{Лемма (Богомолов, Fiorini, \Maks, Pashkovich '15)}%<2->
			Если $t_{i+1} - t_{i} = \text{const}$, $i \in [n-1]$, то
			\[\xc(\CP(2,T_n)) \le 2\lfloor \log_2 (n-1) \rfloor + 2.\]
		\end{alertblock}
		\begin{block}{Теорема (Fiorini, Rothvo\ss, Tiwary '12)}%<3->
			Существует $T_n \subset [2n]$ такое, что\\
			\[\xc(\CP(2,T_n)) = \Omega(\sqrt{n/\log n}).\]
		\end{block}
	\end{minipage}
	\hfill
	\begin{minipage}[t]{0.37\textwidth}
		\begin{exampleblock}{}
			\centering

		\end{exampleblock}
	\end{minipage}
	
	\begin{block}{Теорема (Ben-Tal\&Nemirovski '01)}%<3->
		Сложность расширения правильного $n$-угольника равна~$\Theta(\log n)$.
	\end{block}
	%	\visible<3->{

	%	\end{block}

\end{frame}

\begin{frame}{Циклический многогранник: сложность расширения}
	\begin{alertblock}{Теорема (Богомолов, Fiorini, \Maks, Pashkovich '15)}
		%Если расстояния между соседними числами в~$T_n$ одинаковы, то%\\
		Если $t_{i+1} - t_{i} = \text{const}$, $i \in [n-1]$, то
		%Пусть $S_N = \{1,2,\dots,N\}$. Тогда
		%		\centering
		\[\alert{\xc}\bigl(\CP(d, T_n)\bigr) \le 2\Bigl(\xc\bigl(\CP(2, T_n)\bigr)\Bigr)^{\lfloor d/2\rfloor} = \Theta\bigl(\log n\bigr)^{\lfloor d/2\rfloor}.\]
	\end{alertblock}
	\begin{block}{Теорема (Fiorini, Kaibel, Pashkovich, Theis '12)}%<2->
		Оценка сверху для числа прямоугольного покрытия:%\\
		%		\centering
		\[\alert{\rc}\bigl(\CP(d, T_n)\bigr) = \bigO\bigl(d^2\log n\bigr).\]
	\end{block}
\end{frame}

		%и~\textbf{полиномиальным} числом прямоугольного покрытия.\\

	\end{block}

		Имеются два семейства конечных множеств $X_n\subset \Z^n_+$ и $Y_n\subset \Z^n_+$: %, $n\in \N$:

			\item $\conv(X_n)$ и $\conv(Y_n)$ \textbf{комбинаторно эквивалентны};
			\item задача оптимизации на $\conv(X_n)$ \textbf{полиномиально разрешима}, % по $n$.

		\end{enumerate}

\begin{frame}{Многогранник орпутей}
	
	\begin{block}{Обозначения}
		Пусть $\alert{D_n} = (V,A)$ --- \alert{полный орграф}, $|V| = n$, $|A| = n(n-1)$,\\
		$\alert{\vec{x}} = \alert{\vec{x}(S)} \in \{0,1\}^A$ --- \alert{характеристический вектор} подмножества $S \subseteq A$:\\
		для каждой дуги $a \in A$, если $a \in S$, то $x_{a} = 1$, иначе $x_{a} = 0$.
	\end{block}
	
	\begin{block}{Многогранник орпутей}
		\centering
		\(
		\alert{\Dipath(n)} \coloneqq \conv\bigl\{\vec{x}(S) \bigm| \text{$S$ --- путь в орграфе $D_n$ из $v_1$ в $v_n$}\bigr\}
		\)
	\end{block}
	\begin{block}{Замечание}<2->
		Многогранник гамильтоновых контуров $\ATSP(n)$ аффинно эквивалентен грани многогранника орпутей $\Dipath(n+1)$.
		
		Как следствие, $\Dipath(n+1)$ имеет сверхполиномиальные\\ характеристики сложности: кликовое число графа,\\ число прямоугольного покрытия, \dots
	\end{block}
\end{frame}

\begin{frame}{Полиэдр кратчайших орпутей}
	\begin{block}{Условие задачи о кратчайшем пути}
		В орграфе $D_n$ отсутствуют контуры отрицательной длины.
	\end{block}
	\begin{block}{Определение}
		Пусть $K(n)  \coloneqq \cone\bigl\{\vec{x}(S) \bigm| \text{$S$ --- контур в орграфе $D_n$}\bigr\}$.\\
		\alert{Полиэдр кратчайших орпутей} есть сумма Минковского
		\[
		\alert{\ShortP(n)} \coloneqq \Dipath(n) + K(n).
		\]
	\end{block}
	\begin{block}{Утверждение (Saigal '69)}
		Полиэдр $\ShortP(n)$ имеет простое H-описание,\\
		состоящее из $n(n-1)$ линейных неравенств и $n$ уравнений.
	\end{block}
	\begin{alertblock}{Теорема (\Maks '04)}
		Кликовое число графа полиэдра $\ShortP(n)$ равно $\lfloor n^2 / 4\rfloor$.
	\end{alertblock}
\end{frame}


\begin{frame}{Основные результаты (кратко)}
	\begin{enumerate}
		\item Диаметр графа циклического многогранника.
		\item Сложность расширения циклического многогранника.
		\item Кликовое число полиэдра кратчайших орпутей.
		\item Многогранники Мацуи являются гранями многогранников с~NP-полным критерием несмежности вершин.
		\item $\BQP(n)$ --- общая грань многогранников NP-трудных задач.
		\item $\BQP(n)$ содержит $k$-смежностные грани со сверхполиномиальным числом вершин.
		\item Аналог теоремы Кука для многогранников.
		\item Контрпример для кликового числа графа многогранника и примеры алгоритмов непрямого типа.
		\item Контрпример для числа прямоугольного покрытия.
		\item Контрпример для произвольной комбинаторной характеристики.
	\end{enumerate}
\end{frame}

\begin{frame}{Сложность задач и характеристики многогранника}

			\item \textbf{Число вершин}~--- верхняя оценка сложности в переборном алгоритме.
			\item \textbf{Число гиперграней}~--- оценка сложности задачи линейного программирования.
			\item Различные \textbf{характеристики графа} многогранника: \\
			диаметр, максимальная степень вершины, кликовое число графа, \dots
			\item Сложность проверки произвольной пары вершин на смежность.
			\item \textbf{Сложность расширения} многогранника.
			\item \dots
			%     \item<5-> Минимальное число гиперграней расширенной формулировки многогранника.
			%		
			%				(Многогранник $Q$ называется расширенной 				
			%				формулировкой многогранника $P$, 
			%

	\begin{block}{Пример: многогранник коммивояжера $TSP_n$}
		\textbf{Размерность} равна $n (n-3)/2$, \textbf{число вершин} равно $(n-1)!/2$,
		\textbf{число гиперграней} сверхэкспоненциально, \textbf{диаметр} равен двум, 
		\textbf{кликовое число графа} не меньше $2^{\sqrt{n}/2}$, 
		задача проверки вершин на не\textbf{смежность} NP-полна,
		\textbf{сложность расширенной формулировки} $\xc(\TSP_n) = 2^{\Omega(n)}$.
	\end{block}

\begin{frame}{\hypertarget{BondComments}{О} широком классе алгоритмов 1}
	\begin{block}{Бондаренко, Николаев '13}
	Известно, что трудоемкость алгоритмов прямого типа, использующих
линейные сравнения, по поиску максимума линейной целевой функции $(c, x)$ на множестве $X$,
или, что то же самое, нахождения конуса $K(x)$, которому принадлежит целевой вектор $c$, не может
быть меньше значения $\omega(X)$. Таким образом, $\omega(X)$ является некоторой условной характеристикой
сложности задач комбинаторной оптимизации.
	\end{block}

	\begin{block}{Антонов, Бондаренко '12}
В последние годы появилось большое количество работ, в которых устанавливается, 
что вычислительную сложность задач комбинаторной оптимизации отражают некоторые свойства
графов многогранников, порождаемых этими задачами.
В частности, заметную роль играет плотность графов многогранников,
которая служит нижней границей временной трудоемкости алгоритмов из широкого класса,
включающего большинство известных методов.
	\end{block}
\end{frame}

\begin{frame}{О широком классе алгоритмов 2}
	\begin{block}{Бродский '09}
Как известно, вычислительную сложность задач комбинаторной оптимизации отражают
некоторые свойства графов многогранников, порождаемых этими задачами.
В частности, заметную роль играет плотность графов многогранников,
которая служит нижней границей временной трудоемкости алгоритмов из широкого класса,
включающего большинство известных комбинаторных методов. 
	\end{block}

	\begin{block}{Бондаренко, Бродский '08}
Среди числовых характеристик полиэдральных графов, отражающих различные аспекты вычислительной
сложности соответствующих задач, особое место принадлежит плотности графа. 
В частности, оценки плотности полиэдральных графов большого количества комбинаторных
задач, полученные в работах [3–18], показали, что эта характеристика экспоненциальна
по размерности многогранников для труднорешаемых задач и полиномиальна для полиномиально разрешимых.
	\end{block}
