\begin{frame}{Анализ общей формулировки}
	\begin{block}{Определение (неформальное)}
		\alert{Задача комбинаторной оптимизации} --- это выбор оптимального\\ решения из~некоторого конечного множества (допустимых решений).\\
		\alert{Массовая задача} --- параметры не зафиксированы.\\
		\alert{Индивидуальная задача} --- все входные данные зафиксированы.\\
		\alert{Код задачи} --- набор входных данных, от которых зависит множество допустимых решений.\\
	\end{block}
	\begin{block}{Пример: задача о рюкзаке}
		\alert{Допустимое решение} --- набор предметов, помещающихся в рюкзак.\\
		Множество допустимых решений зависит от (\alert{кода задачи}):\\
		числа предметов, их~размеров и размера рюкзака.\\
		От ценностей предметов зависит только выбор оптимального решения.
		Набор ценностей --- \alert{целевой вектор} задачи о рюкзаке.
	\end{block}
\end{frame}

\begin{frame}{Линейная задача комбинаторной оптимизации}
\begin{block}{Определение}
\textbf{Линейная задача комбинаторной оптимизации (ЛЗКО)} "---\\
это~тройка функций, аргументом которых является \alert{код задачи $I$}:
\begin{enumerate}
	\item \alert{Размерность} задачи $\alert{d} = d(I) \in \N$.
	\item Функция $k = k(I) \in \N$, задающая \alert{пространство решений} $\alert{U} \coloneqq \{0,\dots,k\}^d$. В большинстве примеров $k\equiv 1$.
	\item \alert{Предикат допустимости} $\alert{g(\bm{x}, I)} \in \{\text{ложь}, \text{истина}\}$, где $\bm{x} \in U$.
\end{enumerate}
\medskip
\textbf{Входные данные:} код задачи $I$ и \alert{целевой вектор} $\alert{\bm{c}} \in \Z^d$.

\medskip
\textbf{Цель:} \alert{$\langle\bm{c},\bm{x}\rangle \to \max$} при условии $\bm{x} \in X$,\\
%найти оптимальное решение
%\[\bm{x^*} = \arg\max_{\bm{x} \in X} \langle\bm{c},\bm{x}\rangle,\]
где
$X = X(I) \coloneqq \{\bm{x} \in U \mid g(\bm{x}, I)\}$ "--- множество \alert{допустимых} решений.
\end{block}
\end{frame}

\begin{frame}{Пример 1}
\begin{block}{Задача о кратчайшем пути}
\begin{description}
	\item[Дано:] полный граф $K_n = ([n],E)$, $[n] \coloneqq \{1,\dots,n\}$;\\
	длина $c_e \in \Z$ для каждого ребра $e \in E$.
	\item[Найти:] кратчайший путь из вершины $1$ в вершину $n$.
\end{description}
\end{block}

\begin{block}{Задача о кратчайшем пути в виде ЛЗКО}
\begin{enumerate}
\item \textbf{Код задачи} "--- число вершин графа $K_n$.
\item \textbf{Размерность} задачи равна числу ребер графа.
\item \textbf{Пространство решений} $U = \{0,1\}^E$ \\
состоит из характеристических векторов подграфов графа $K_n$.
\item \textbf{Предикат допустимости} $g = g(\bm{x}, n)$ равен <<истина>>,\\
если $\bm{x} \in \{0,1\}^E$ "--- характеристический вектор пути\\
в $K_n$ из вершины $1$ в вершину $n$.
\end{enumerate}
\textbf{Входные данные:} число вершин $n$ и вектор длин ребер $\bm{c} = (c_e) \in \Z^E$.
\end{block}
\end{frame}

\begin{frame}{Пример 2}
\begin{block}{Булево квадратичное программирование}
\begin{description}
	\item[Дано:] число $n$ булевых переменных $x_i \in \{0,1\}$, $i \in [n]$;\\ матрица коэффициентов $\bm{c} = (c_{ij}) \in \Z^{n \times n}$.
	\item[Найти:] $\max \displaystyle\sum_{i,j \in [n]} c_{ij} x_i x_j$.
\end{description}
\end{block}
	
\begin{block}{Булево квадратичное программирование в виде ЛЗКО}
	\begin{enumerate}
		\item \textbf{Код задачи} "--- число переменных $n$.
		\item \textbf{Размерность} задачи равна $n^2$.
		\item \textbf{Пространство решений} $U = \{0,1\}^{n \times n}$.
		\item \textbf{Предикат допустимости} $g = g(\bm{x}, n)$ равен <<истина>>,\\
		если $\bm{x} = (x_{ij}) \in \{0,1\}^{n \times n}$ и \alert{$x_{ij} = x_{ii} x_{jj}$}.
	\end{enumerate}
	\textbf{Входные данные:} число $n$ и матрица $\bm{c} \in \Z^{n \times n}$.
\end{block}
\end{frame}

\begin{frame}{Многогранники задачи}
\begin{block}{Определение}
При оптимизации $\langle\bm{c},\bm{x}\rangle \to \max$ \\
множество допустимых решений $X = X(I) \subseteq \{0,\dots,k\}^d$\\
можно заменить его выпуклой оболочкой $\alert{P(I) \coloneqq \conv(X(I))}$,\\
которая называется \alert{многогранником задачи}.\\
При произвольном выборе кода $I$ получаем \alert{семейство многогранников}.
\end{block}

\begin{block}{Пример: булев квадратичный многогранник}
\[\alert{\BQP(n)} \coloneqq \conv \Bigl\{ \bm{x} = (x_{ij}) \in \{0,1\}^{n \times n} \Bigm| x_{ij} = x_{ii} x_{jj}, \ i,j\in[n] \Bigr\}\]
\end{block}

\begin{block}{Для чего нужны многогранники задач?}
	\begin{enumerate}
	\item Использование геометрических методов для решения ЛЗКО.
	\item Явное представление структуры множества допустимых решений.
	\end{enumerate}
\end{block}
\end{frame}


\begin{frame}{ЛЗКО и задача линейного программирования}
	\begin{block}{ЛЗКО}
		$\langle\bm{c},\bm{x}\rangle \to \max$ при условии \alert{$\bm{x} \in \conv(X(I))$},\\
		где %$X(I) \subseteq \{0,\dots,k\}^d$ "--- множество допустимых решений,\\
		$\conv(X(I))$ "--- многогранник задачи.
	\end{block}

	\begin{block}{Задача линейного программирования}
		$\langle\bm{c},\bm{x}\rangle \to \max$ при условии \alert{$A \bm{x} \le \bm{b}$},
		где $\bm{x} \in \R^d$, $A \in \Z^{n \times d}$, $\bm{b} \in \Z^n$.
	\end{block}
	\begin{block}{Способы описания выпуклого многогранника}
		\begin{description}
		\item[V-описание:] выпуклая оболочка конечного множества точек.
		\item[H-описание:] пересечение полупространств.
		\end{description}
		\textbf{Пример} <<0/1-куб>>: $\conv\left(\{0,1\}^d\right) = \{\bm{x} \in \R^d \mid \bm{0} \le \bm{x} \le \bm{1}\}$.
	\end{block}
\end{frame}

\begin{frame}{Смежные решения}
	\begin{block}{Определение}
		Для задачи линейной оптимизации на конечном множестве $X \subset \Z^d$\\
		пару точек $\bm{x}, \bm{y} \in X$ назовем \alert{смежными},\\
		если найдется $\bm{c_{xy}} \in \Z^d$ такой, что 
		\[\langle\bm{c_{xy}}, \bm{x}\rangle = \langle\bm{c_{xy}}, \bm{y}\rangle > \langle\bm{c_{xy}}, \bm{z}\rangle \qquad 
		\forall \bm{z} \in X \setminus \{\bm{x}, \bm{y}\}.\]
		Смежность $\bm{x}$ и $\bm{y}$ означает, что $\bm{x}$ и $\bm{y}$ являются вершинами многогранника $\conv(X)$ и образуют \alert{ребро} этого многогранника.
	\end{block}
	\begin{block}{Наблюдение}
		Алгоритм для линейной оптимизации на $X$ должен быть чувствителен к небольшим изменениям вектора $\bm{c_{xy}}$, так как оптимальное решение может меняться с $\bm{x}$ на $\bm{y}$ и обратно.
		
		Таким образом, наличие ребер (1-граней) между вершинами многогранника влияет на структуру алгоритма.
	\end{block}
\end{frame}


\begin{frame}{Примеры характеристик сложности для многогранников}
	\begin{block}{Основные характеристики}

			\item \alert{Размерность $\dim(P)$} --- нижняя оценка сложности задачи.
			\item \alert{Число вершин $\vertices(P)$} --- верхняя оценка сложности в~переборном алгоритме.
			\item \alert{Число гиперграней $\facet(P)$} --- оценка сложности задачи линейного программирования.
%			\item Различные \textbf{характеристики графа} многогранника: \\
%			диаметр, кликовое число, \dots
%			\item Сложность проверки произвольной пары вершин на смежность.
		\end{enumerate}
	\end{block}
	
	\begin{block}{Пример: булев квадратичный многогранник $\BQP(n)$}
		$\dim(\BQP(n)) = n (n+1)/2$, \\
		$\vertices(\BQP(n)) = 2^n$,\\
		$\facet(\BQP(n))$ сверхэкспоненциально.

	\end{block}

\begin{frame}{Диаметр графа многогранника}
	\begin{block}{Классическая мотивация}
		\alert{Диаметр графа многогранника является нижней оценкой \\
		числа шагов симплекс"=метода (в~худшем случае).}
	\end{block}
	\begin{block}{Гипотеза (Hirsch '57)}
		Диаметр графа выпуклого многогранника не~превышает разности
		
		между числом гиперганей и~размерностью многогранника:
		
		\centerline{\(\diam(P) \le \facet(P) - \dim(P)\).}
	\end{block}
	
	\begin{block}{Справка}
	С~постановки гипотезы до~настоящего времени \\
	опубликованы десятки высокорейтинговых работ,\\
	содержащих оценки диаметров графов многогранников.
	\end{block}
\end{frame}

\begin{frame}{Матрица инциденций вершин-гиперграней}

		\end{cases}

\begin{frame}{Число прямоугольного покрытия}
	\begin{block}{Число прямоугольного покрытия матрицы инциденций $M$}
		\centering
		\(\alert{\rc(M)} \coloneqq \min\{\text{\# 0-прямоугольников для покрытия всех нулей в $M$}\}\)
	\end{block}
	
	\begin{block}{Теорема (Yannakakis '91)}

		числом прямоугольного покрытия матрицы инциденций $M$:\\
		\centerline{\(\rc(M) \le \xc(P)\).}
	\end{block}

		а $f(P)$~--- число его граней всех размерностей. Тогда
		\centerline{\(\dim(P) < \log_2 f(P) \le \alert{\rc(M) \le \xc(P)} \le \min\{\vertices(P), \facet(P)\}\)}
	\end{block}

\begin{frame}{Циклический многогранник}
\begin{block}{Определение}
	Пусть $T = \{t_1, \dots, t_n\} \subset \R$, $t_1 < t_2 < \dots < t_n$.\\
	\alert{Циклическим многогранником} называется выпуклая оболочка
	\[\alert{\CP(d,T)} \coloneqq \conv\bigl\{(t, t^2, \dots, t^d) \bigm| t \in T\bigr\}, \qquad 2 \le d < n,\]
	$d$ --- размерность многогранника,
	$n$ --- число его вершин.
\end{block}
	
\begin{block}{Теорема, McMullen '70 (гипотеза, Motzkin '57)}
	Циклический многогранник~$\CP(d,T)$ для любого $k \in [d]$ обладает\\ \alert{максимальным числом $k$-граней} среди всех выпуклых многогранников той же размерности и с~тем~же числом вершин.
	
	В частности, он $\lfloor d/2\rfloor$-смежностен.
\end{block}
\end{frame}

	среди всех $d$-многогранников с~тем~же числом гиперграней.
\end{block}

\begin{frame}{Многогранник орпутей}
\begin{block}{Определение}
	Пусть $G = (V,A)$ --- полный орграф, $|V| = n$, $|A| = n(n-1)$,\\
	$\bm{x} = \bm{x}(S) \in \{0,1\}^A$ --- характеристический вектор подмножества $S \subseteq A$:\\
	\centerline{
	\(x_{a}(S) \coloneqq \begin{cases}
		1,& \text{если $a \in S$,}\\
		0,& \text{иначе.}
	\end{cases}
	\)}
	\alert{Многогранником орпутей} называется выпуклая оболочка
	\[
	\alert{\Dipath(n)} \coloneqq \conv\bigl\{\bm{x}(S) \bigm| \text{$S$ --- путь в орграфе $G$ из $v_1$ в $v_n$}\bigr\}.
	\]
\end{block}
\begin{block}{Утверждение}
	Многогранник задачи коммивояжера $\ATSP(n)$ аффинно эквивалентен грани многогранника орпутей $\Dipath(n+1)$.
	
	Как следствие, $\Dipath(n+1)$ имеет большие (сверхполиномиальные) характеристики сложности: кликовое число графа,\\ число прямоугольного покрытия, \dots
\end{block}
\end{frame}

\begin{frame}{Полиэдр кратчайших орпутей}
\begin{block}{Определение}
	Пусть $K(n)  \coloneqq \cone\bigl\{\bm{x}(S) \bigm| \text{$S$ --- контур в орграфе $G$}\bigr\}$.\\
	\alert{Полиэдр кратчайших орпутей} есть сумма Минковского
	\[
	\alert{\ShortP(n)} \coloneqq \Dipath(n) + K(n).
	\]
\end{block}
\begin{block}{Утверждение (Saigal '69)}
	Полиэдр $\ShortP(n)$ имеет простое H-описание,\\
	состоящее из $n(n-1)$ линейных неравенств и $n$ уравнений.
\end{block}
\begin{alertblock}{Теорема (\Maks '04)}
	Кликовое число графа полиэдра $\ShortP(n)$ равно $\lfloor n^2 / 4\rfloor$.
\end{alertblock}
\end{frame}

\section{Аффинная сводимость}

\begin{frame}{Аффинная сводимость многогранников задач}
\begin{block}{Определение}
	Пусть $P = \{p(I)\}$ и $Q = \{q(I')\}$ --- семейства многогранников.\\
	Семейство $P$ \alert{аффинно сводится} к $Q$ (обозначение: \alert{$P \propto_A Q$}),\\ если найдутся полиномиально вычислимые: %(относительно размера многогранника $p\in P$):
	\begin{enumerate}
		\item 
		Преобразование $\tau$ кода $I$ каждого многогранника $p = p(I)\in P$ в~код $I'$ многогранника $q = q(I') \in Q$.
		\item 
		Алгоритм построения для каждого кода $I$ аффинного отображения 
		\[
		\alpha\from \R^d \to \R^{d'}, \qquad d = d(I), \quad d' = d'(\tau(I)),
		\]
		такого, что многогранник \alert{$\alpha(p(I))$ аффинно эквивалентен $p(I)$\\ 
		и является гранью} (возможно несобственной) многогр-ка \alert{$q(\tau(I))$}.
	\end{enumerate}
\end{block}
\end{frame}

\begin{frame}{Грань с NP-полным критерием несмежности}
	\begin{alertblock}{Теорема (\Maks '12)}%<2->
		Все вышеперечисленные многогранники содержат в~качестве граней\\ многогранники двойных покрытий (многогранники Мацуи),\\
		для которых задача проверки на~несмежность NP-полна.
	\end{alertblock}
	\begin{block}{Пример 1}
		Все вершины булева квадратичного многогранника попарно смежны.
	\end{block}
	\begin{block}{Пример 2}
	Пусть $G=(V,E)$ --- произвольный граф на $n = |V|$ вершинах.
	\alert{Многогранником независимых множеств} называется
	\centerline{\(
		\alert{\Stable(G)} \coloneqq \conv\bigl\{\bm{x}\in\{0,1\}^V \bigm| x_v + x_u \le 1 \quad \forall \{v,u\} \in E\bigr\}.
		\)}
	У этого многогранника имеется простой критерий смежности вершин.
	%пара вершин многогранника независимых множеств $\Stable(G)$ смежна тогда и только тогда, когда симметрическая разность соответствующих	независимых множеств индуцирует связный подграф в $G$.
	\end{block}

\begin{frame}{Многогранники с простым критерием смежности}
	\begin{alertblock}{Теорема (\Maks '17)}
		Многогранники независимых множеств $\Stable(G)$\\
		аффинно сводятся к многогранникам Мацуи.\\
		Но, многогранники Мацуи (за исключением отрезков)\\
		не могут быть гранями многогранника $\Stable(G)$ ни при каком $G$.
	\end{alertblock}
	\begin{alertblock}{Теорема (\Maks '15)}%<2->
		Многогранники независимых множеств эквивалентны относительно аффинной сводимости многогранникам следующих задач:\\
		упаковки и~разбиения множества, 3-сочетания, раскраска графа.
	\end{alertblock}
\end{frame}

\begin{frame}{Сверхполиномиальность характеристик}
	\begin{alertblock}{Теорема (\Maks)}
	Семейство булевых квадратичных многогранников $\{\BQP_n\}$\\
	аффинно сводится к каждому из перечисленных выше\\
	семейств многогранников труднорешаемых задач,
	а также к~многогранникам следующих задач: линейное упорядочивание,\\
	задача Штейнера в~графе и~некоторых других.\\
	\textbf{Замечание:} аффинное свед\'ение в обратную сторону невозможно.
	\end{alertblock}
	\begin{block}{Теорема (Fiorini, Massar, Pokutta, Tiwary, de Wolf '11)}
	Число прямоугольного покрытия (а~значит и~сложность расширения)
	
	для~булева квадратичного многогранника экспоненциально.
	\end{block}
	\begin{block}{Теорема (Deza, Laurent, Poljak '92)}
	Булев квадратичный многогранник $\BQP_n$ 3-смежностен.\\
	Следовательно, кликовое число графа совпадает с числом вершин $2^n$.
	\end{block}
\end{frame}


\begin{frame}{Универсальные $k$-смежностные грани}
	\begin{block}{Вопрос}
		Для каких значений $k$ булевы квадратичные многогранники \\
		содержат $k$-смежностные грани со сверхполиномиальным числом вершин?

	\begin{alertblock}{Теорема (\Maks '14)}%<2->

		содержат $k$-смежностные грани со~сверхполиномиальным \\
		(относительно размерности многогранника) числом вершин.
	\end{alertblock}
\end{frame}

\begin{frame}{Аналог теоремы Кука для многогранников}
	\begin{block}{Аффинная сводимость}
		Для каждого многогранника $p(I) \in P$ определен многогранник $q(I') \in Q$ такой, что $p(I)$ \alert{аффинно эквивалентен} грани $q(I')$.
	\end{block}
	\begin{block}{Расширенная аффинная сводимость}
		Для каждого многогранника $p(I) \in P$ определен многогранник $q(I') \in Q$ такой, что $p(I)$ \alert{является аффинным образом} грани $q(I')$.
	\end{block}
	\begin{alertblock}{Теорема (\Maks '12)}
		Пусть для семейства многогранников $\bigl\{\conv(X(I))\bigr\}$ задача проверки условия $\bm{x} \in X(I)$ (предикат допустимости) принадлежит классу NP.
		Тогда это семейство расширенно аффинно сводится к булевым квадратичным многогранникам.
	\end{alertblock}
\end{frame}

\begin{frame}{Комбинаторика многогранника VS сложность задачи} %{Пример NP-трудной задачи с полиномиальным $\rc()$}
	\begin{block}{Вопрос}
		Выше были представлены примеры \textbf{труднорешаемых} задач с~\textbf{полиномиальным} кликовым числом
		и~\textbf{полиномиальным} числом прямоугольного покрытия.\\
		Можно~ли, зная \textbf{только комбинаторные} свойства многогранника, \\
		оценить сложность соответствующей оптимизационной~задачи?
	\end{block}
	\begin{alertblock}{Теорема (Максименко '16)}<2->
		Существуют два семейства конечных множеств $X_n\subset \Z^n_+$ и $Y_n\subset \Z^n_+$, $n\in \N$:
		%		Существуют два семейства многогранников $\conv(X_n)$ и $\conv(Y_n)$, $n\in \N$:
		\begin{enumerate}
			\item многогранники $\conv(X_n)$ и $\conv(Y_n)$ \textbf{комбинаторно эквивалентны};
			%			\item $\conv(X_n)$ и $\conv(Y_n)$ \textbf{комбинаторно эквивалентны},
			\item $\dim(X_n) = n$, $|X_n| = 2^n$;
			\item для любого $x\in X_n$ и соответствующего ему $y \in Y_n$,\\
			длины их двоичных записей ограничены сверху полиномом от $n$\\
			и, кроме того, $0 \le \frac{\|x-y\|_{\infty}}{\|x\|_{\infty}} < \frac{1}{2^n}$.
			\item Задача оптимизации на $\conv(X_n)$ \textbf{полиномиально разрешима} по $n$.
			\item Задача оптимизации на $\conv(Y_n)$ \textbf{экспоненциально сложна}.
		\end{enumerate}
	\end{alertblock}

\begin{frame}{Сложность задач и характеристики многогранника}
	\begin{block}{Примеры характеристик сложности для многогранников}
		\begin{enumerate}
			\item \textbf{Размерность} многогранника~--- нижняя оценка сложности задачи.
			\item \textbf{Число вершин}~--- верхняя оценка сложности в переборном алгоритме.
			\item \textbf{Число гиперграней}~--- оценка сложности задачи линейного программирования.
			\item Различные \textbf{характеристики графа} многогранника: \\
			диаметр, максимальная степень вершины, кликовое число графа, \dots
			\item Сложность проверки произвольной пары вершин на смежность.
			\item \textbf{Сложность расширения} многогранника.
			\item \dots
			%     \item<5-> Минимальное число гиперграней расширенной формулировки многогранника.
			%		
			%				(Многогранник $Q$ называется расширенной 				
			%				формулировкой многогранника $P$, 
			%				
			%				если существует аффинное отображение $\pi$: $\pi(Q) = P$.)
			%     \item<6-> \dots
		\end{enumerate}

	\begin{block}{Пример: многогранник коммивояжера $TSP_n$}
		\textbf{Размерность} равна $n (n-3)/2$, \textbf{число вершин} равно $(n-1)!/2$,
		\textbf{число гиперграней} сверхэкспоненциально, \textbf{диаметр} равен двум, 
		\textbf{кликовое число графа} не меньше $2^{\sqrt{n}/2}$, 
		задача проверки вершин на не\textbf{смежность} NP-полна,
		\textbf{сложность расширенной формулировки} $\xc(\TSP_n) = 2^{\Omega(n)}$.
	\end{block}

\begin{frame}{\hypertarget{BondComments}{О} широком классе алгоритмов 1}
