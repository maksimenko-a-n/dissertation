\begin{frame}{Линейная задача комбинаторной оптимизации}
\begin{block}{Определение}
\textbf{Линейная задача комбинаторной оптимизации (ЛЗКО)} "--- это~тройка функций:
\begin{enumerate}
	\item \alert{Размерность} задачи $d = d(I) \in \N$, где $I$ "--- код задачи, являющийся частью ее входных данных.
	\item Функция $k = k(I) \in \N$, задающая \alert{пространство решений} $U = \{0,\dots,k\}^d$. В большинстве примеров $k\equiv 1$.
	\item \alert{Предикат допустимости} $g = g(\bm{x}, I) \in \{\text{ложь}, \text{истина}\}$, где $\bm{x} \in U$.
\end{enumerate}
\medskip
\textbf{Входные данные:} код задачи $I$ и целевой вектор $\bm{c} \in \Z^d$.

\medskip
\textbf{Цель:} $\langle\bm{c},\bm{x}\rangle \to \max$ при условии $\bm{x} \in X$,\\
%найти оптимальное решение
%\[\bm{x^*} = \arg\max_{\bm{x} \in X} \langle\bm{c},\bm{x}\rangle,\]
где
$X = X(I) = \{\bm{x} \in U \mid g(\bm{x}, I)\}$ "--- множество \alert{допустимых} решений.
\end{block}
\end{frame}

\begin{frame}{Пример 1}
\begin{block}{Задача о кратчайшем пути}
\begin{description}
	\item[Дано:] полный граф $K_n = ([n],E)$, $[n] = \{1,\dots,n\}$;\\
	длина $c_e \in \Z$ для каждого ребра $e \in E$.
	\item[Найти:] кратчайший путь из вершины $1$ в вершину $n$.
\end{description}
\end{block}

\begin{block}{Задача о кратчайшем пути в виде ЛЗКО}
\begin{enumerate}
\item Код задачи "--- число вершин графа $K_n$.
\item Размерность задачи равна числу ребер графа.
\item Пространство решений $U = \{0,1\}^E$ \\
состоит из характеристических векторов подграфов графа $K_n$.
\item Предикат допустимости $g = g(\bm{x}, n)$ равен <<истина>>,\\
если $\bm{x} \in \{0,1\}^E$ "--- характеристический вектор пути\\
в $K_n$ из вершины $1$ в вершину $n$.
\end{enumerate}
\textbf{Входные данные:} число вершин $n$ и вектор длин ребер $\bm{c} = (c_e) \in \Z^E$.
\end{block}
\end{frame}

\begin{frame}{Пример 2}
\begin{block}{Булево квадратичное программирование}
\begin{description}
	\item[Дано:] число $n$ булевых переменных $x_i \in \{0,1\}$, $i \in [n]$;\\ матрица коэффициентов $\bm{c} = (c_{ij}) \in \Z^{n \times n}$.
	\item[Найти:] $\max \displaystyle\sum_{i,j \in [n]} c_{ij} x_i x_j$.
\end{description}
\end{block}
	
\begin{block}{Булево квадратичное программирование в виде ЛЗКО}
	\begin{enumerate}
		\item Код задачи "--- число переменных $n$.
		\item Размерность задачи равна $n^2$.
		\item Пространство решений $U = \{0,1\}^{n \times n}$.
		\item Предикат допустимости $g = g(\bm{x}, n)$ равен <<истина>>,\\
		если $\bm{x} = (x_{ij}) \in \{0,1\}^{n \times n}$ и \alert{$x_{ij} = x_{ii} x_{jj}$}.
	\end{enumerate}
	\textbf{Входные данные:} число $n$ и матрица $\bm{c} \in \Z^{n \times n}$.
\end{block}
\end{frame}

\begin{frame}{Многогранники задачи}
\begin{block}{Определение}
При оптимизации $\langle\bm{c},\bm{x}\rangle \to \max$ \\
множество допустимых решений $X = X(I) \subseteq \{0,\dots,k\}^d$\\
можно заменить его выпуклой оболочкой $\alert{P(I) = \conv(X(I))}$,\\
которая называется \alert{многогранником задачи}.

При произвольном выборе кода $I$ получаем \alert{семейство многогранников}.
\end{block}

\begin{block}{Пример: булев квадратичный многогранник}
\[\alert{\BQP(n)} = \conv \Bigl\{ \bm{x} = (x_{ij}) \in \{0,1\}^{n \times n} \Bigm| x_{ij} = x_{ii} x_{jj}, \ i,j\in[n] \Bigr\}\]
\end{block}
\end{frame}


\begin{frame}{ЛЗКО и задача линейного программирования}
	\begin{block}{ЛЗКО}
		$\langle\bm{c},\bm{x}\rangle \to \max$ при условии \alert{$\bm{x} \in \conv(X(I))$},\\
		где %$X(I) \subseteq \{0,\dots,k\}^d$ "--- множество допустимых решений,\\
		$\conv(X(I))$ "--- многогранник задачи.
	\end{block}

	\begin{block}{Задача линейного программирования}
		$\langle\bm{c},\bm{x}\rangle \to \max$ при условии \alert{$A \bm{x} \le \bm{b}$},
		где $\bm{x} \in \R^n$, $A \in \Z^{m \times n}$, $\bm{b} \in \Z^m$.
	\end{block}
	\begin{block}{Способы описания выпуклого многогранника}
		\begin{description}
		\item[V-описание:] выпуклая оболочка конечного множества точек.
		\item[H-описание:] пересечение полупространств.
		\end{description}
		\textbf{Пример} <<0/1-куб>>: $\conv\left(\{0,1\}^d\right) = \{\bm{x} \in \R^d \mid \bm{0} \le \bm{x} \le \bm{1}\}$.
	\end{block}
\end{frame}

\begin{frame}{Смежные решения}
	\begin{block}{Определение}
		Для задачи линейной оптимизации на конечном множестве $X \subset \Z^d$\\
		пару векторов $\bm{x}, \bm{y} \in X$ назовем \textbf{смежными},\\
		если найдется $\bm{c_{xy}} \in \Z^d$ такой, что 
		\[\langle\bm{c_{xy}}, \bm{x}\rangle = \langle\bm{c_{xy}}, \bm{y}\rangle > \langle\bm{c_{xy}}, \bm{z}\rangle \qquad 
		\forall \bm{z} \in X \setminus \{\bm{x}, \bm{y}\}.\]
		Смежность $\bm{x}$ и $\bm{y}$ означает, что $\bm{x}$ и $\bm{y}$ являются вершинами многогранника $\conv(X)$ и образуют ребро этого многогранника.
	\end{block}
	\begin{block}{Наблюдение}
		Алгоритм для линейной оптимизации на $X$ должен быть чувствителен к небольшим изменениям вектора $\bm{c_{xy}}$, так как оптимальное решение может меняться с $\bm{x}$ на $\bm{y}$ и обратно.
		
		Таким образом, наличие ребер (1-граней) между вершинами многогранника влияет на структуру алгоритма.
	\end{block}
\end{frame}

	\end{block}

\begin{frame}{Диаметр графа многогранника}
	\begin{block}{Гипотеза (Hirsch '57)}
		Диаметр графа выпуклого многогранника не~превышает разности
		
		между числом гиперганей и~размерностью многогранника:
		
		\centerline{\alert{\(\diam(P) \le \facet(P) - \dim(P)\)}}
	\end{block}
	
	\begin{block}{Классическая мотивация}
		Диаметр графа многогранника является нижней оценкой \\
		числа шагов симплекс"=метода (в~худшем случае).
	\end{block}
	
	\begin{block}{Справка}
	С~постановки гипотезы до~настоящего времени \\
	опубликованы десятки высокорейтинговых работ,\\
	содержащих оценки диаметров графов многогранников.
	\end{block}
\end{frame}

\begin{frame}{Матрица инциденций вершин-гиперграней}
	\begin{block}{Определение}
		Пусть $k = \vertices(P)$ --- число вершин многогранника $P$, $n = \facet(P)$.\\
		Тогда матрица инциденций $M\in \{0,1\}^{k \times n}$ определяется так:
		%\(M_{ij} = 1\), если $j$-я вершина принадлежит $i$-ой гиперграни, \(M_{ij} = 0\) иначе.
		\(M_{ij} = \begin{cases}
		1,& \text{если $i$-я вершина принадлежит $j$-ой гиперграни,}\\
		0,& \text{иначе.}
		\end{cases}
		\)\\
		Матрица инциденций содержит в~себе всю~информацию \\
		о~комбинаторной структуре (<<скелете>>) многогранника.
	\end{block}
\vspace{-15pt}
	\begin{minipage}[t]{0.58\textwidth}
	\begin{block}{0-прямоугольник}
		Пусть $M \in \{0,1\}^{k\times n}$~--- матрица инциденций
		вершин-гиперграней.\\
		Набор номеров вершин~$I \subseteq [k]$ \\
		и~номеров гиперграней~$J \subseteq [n]$ \\
		образует \alert{0-прямоугольник~$I\times J$} в~$M$,\\
		если \alert{$M_{ij} = 0$}\quad $\forall$ $i\in I$, $j\in J$.

	\end{block}

\begin{frame}{Число прямоугольного покрытия}
	\begin{block}{Число прямоугольного покрытия}
		\centering
		\(\alert{\rc(M)} = \min\{\text{\# 0-прямоугольников для покрытия всех нулей в $M$}\}\)
	\end{block}
	
	\begin{alertblock}{Теорема (Yannakakis '91)}
		Сложность расширения многогранника оценивается снизу\\
		числом прямоугольного покрытия матрицы инциденций:\\
		\centerline{\(\rc(M) \le \xc(P)\).}
	\end{alertblock}
	\begin{block}{Основные соотношения}
		Пусть $M$~--- матрица инциденций вершин"=гиперграней \\
		многогранника $P$, а $f(P)$~--- число его граней всех размерностей. Тогда
		\centerline{\(\dim(P) < \log_2 f(P) \le \alert{\rc(M) \le \xc(P)} \le \min\{\vertices(P), \facet(P)\}\)}
	\end{block}

\begin{frame}{Нетривиальные характеристики многогранника}
	\begin{block}{}
		\begin{enumerate}
			\item \textbf{Диаметр графа} --- нижняя оценка сложности числа шагов симплекс-метода.
			\item \textbf{Сложность проверки смежности вершин}.
			\item \textbf{Кликовое число графа} --- характеристика сложности в некотором классе алгоритмов.
			\item \textbf{Сложность расширения} --- верхняя оценка сложности задачи линейного программирования.
			\item \textbf{Число прямоугольного покрытия\\ матрицы инциденций вершин-гиперграней}.
		\end{enumerate}
	\end{block}
	
	\begin{block}{Пример: многогранник гамильтоновых циклов $\TSP(n)$}
	%\textbf{Размерность} равна $n (n-3)/2$, \textbf{число вершин} равно $(n-1)!/2$,

	\textbf{число прямоугольного покрытия} $\rc(\TSP(n)) = 2^{\Omega(\sqrt{n})}$.
	\end{block}
\end{frame}

\begin{frame}{\hypertarget{BondComments}{О} широком классе алгоритмов 1}
	\begin{block}{Бондаренко, Николаев '13}
	Известно, что трудоемкость алгоритмов прямого типа, использующих
линейные сравнения, по поиску максимума линейной целевой функции $(c, x)$ на множестве $X$,
или, что то же самое, нахождения конуса $K(x)$, которому принадлежит целевой вектор $c$, не может
быть меньше значения $\omega(X)$. Таким образом, $\omega(X)$ является некоторой условной характеристикой
сложности задач комбинаторной оптимизации.
	\end{block}

графов многогранников, порождаемых этими задачами.
В частности, заметную роль играет плотность графов многогранников,

	\end{block}
